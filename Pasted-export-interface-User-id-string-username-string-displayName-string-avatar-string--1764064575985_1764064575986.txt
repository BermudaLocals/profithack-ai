export interface User {
  id: string;
  username: string;
  displayName: string;
  avatar: string;
  followers: number;
  following: number;
  likes: number;
  bio: string;
  isFollowing?: boolean;
  isVerified?: boolean;
}

export interface Video {
  id: string;
  user: User;
  videoUrl: string;
  thumbnailUrl: string;
  caption: string;
  likes: number;
  comments: number;
  shares: number;
  views: number;
  isLiked?: boolean;
  isBookmarked?: boolean;
  isPremium?: boolean;
  hashtags: string[];
}

export interface Message {
  id: string;
  user: User;
  lastMessage: string;
  timestamp: string;
  unread: number;
  isOnline: boolean;
}

export interface AITool {
  id: string;
  title: string;
  description: string;
  icon: string;
  gradient: [string, string];
}
