# --- DB (users, jobs) ---
class User(SQLModel, table=True):
    id: int | None = Field(default=None, primary_key=True)
    email: str

class Job(SQLModel, table=True):
    id: int | None = Field(default=None, primary_key=True)
    user_email: str
    plugin: str
    status: str = "created"
    input_meta: str = ""
    output_url: str | None = None

engine = create_engine("sqlite:///app.db")
SQLModel.metadata.create_all(engine)

# --- Plugin registry ---
PLUGINS = {}
for path in glob.glob("plugins/*/plugin.py"):
    mod = importlib.import_module(path.replace("/", ".").rstrip(".py"))
    PLUGINS[mod.PLUGIN_META["slug"]] = mod

def render(tpl, **ctx):
    return HTMLResponse(env.get_template(tpl).render(**ctx))

@app.get("/", response_class=HTMLResponse)
async def home(request: Request):
    return render("home.html", plugins=[p for p in PLUGINS.values()])

@app.post("/magic")
async def magic(email: str = Form(...)):
    token = serializer.dumps({"email": email})
    resp = RedirectResponse(url=f"/dash?token={token}", status_code=302)
    return resp

@app.get("/dash", response_class=HTMLResponse)
async def dash(token: str):
    data = serializer.loads(token)
    email = data["email"]
    with Session(engine) as s:
        if not s.exec(SQLModel.select(User).where(User.email==email)).first():
            s.add(User(email=email)); s.commit()
    return render("dash.html", email=email, plugins=PLUGINS)

@app.post("/checkout")
async def checkout():
    if not STRIPE_PRICE_ID:
        return {"url": "/"}
    session = stripe.checkout.Session.create(
        mode="subscription",
        line_items=[{"price": STRIPE_PRICE_ID, "quantity":1}],
        success_url="{YOUR_DOMAIN}/?ok=1",
        cancel_url="{YOUR_DOMAIN}/?canceled=1",
    )
    return {"url": session.url}

@app.post("/run/{slug}")
async def run_plugin(slug: str, request: Request):
    form = await request.form()
    files = [v for k,v in form.multi_items() if isinstance(v, UploadFile)]
    fields = {k:v for k,v in form.items() if not isinstance(v, UploadFile)}
    plugin = PLUGINS[slug]
    output = await plugin.run(fields, files, LLM_API_KEY)
    # save minimal job log
    with Session(engine) as s:
        s.add(Job(user_email=fields.get("email",""), plugin=slug, input_meta=str(fields), status="done", output_url=output.get("url")))
        s.commit()
    return output

# Basic defaults & styles
W, H = 1080, 1920
SAFE_MARGIN = 64  # safe text padding
FONT_PATHS = [
  "/usr/share/fonts/truetype/dejavu/DejaVuSans-Bold.ttf",
  "/usr/share/fonts/truetype/dejavu/DejaVuSans.ttf"

def _font(size=72, bold=True):
  path = FONT_PATHS[0 if bold else 1]
  try:
    return ImageFont.truetype(path, size=size)
  except:
    return ImageFont.load_default()

def wrap_text(s, width=18):
  return "\n".join(textwrap.wrap(s, width=width, break_long_words=False))

def neon_palette():
  # Dark bg + neon accents (no external assets)
  return {
    "bg": (10, 10, 16),
    "fg1": (255, 245, 0),     # neon yellow
    "fg2": (0, 255, 255),     # cyan
    "fg3": (255, 0, 255),     # magenta
    "fg4": (255, 128, 0),     # orange
    "muted": (220, 220, 220)
  }

def gradient_bg():
  # simple vertical gradient
  img = Image.new("RGB", (W, H), (10,10,16))
  top = (10, 10, 16)
  bottom = (24, 24, 40)
  for y in range(H):
    r = int(top[0] + (bottom[0]-top[0]) * y / H)
    g = int(top[1] + (bottom[1]-top[1]) * y / H)
    b = int(top[2] + (bottom[2]-top[2]) * y / H)
    for x in range(W):
      img.putpixel((x,y), (r,g,b))
  return img

def draw_centered_text(draw, text, y, color, size, bold=True, shadow=True):
  f = _font(size, bold)
  w, h = draw.multiline_textbbox((0,0), text, font=f, align="center")[2:]
  x = (W - w)//2
  if shadow:
    draw.multiline_text((x+4, y+4), text, font=f, align="center", fill=(0,0,0))
  draw.multiline_text((x, y), text, font=f, align="center", fill=color)

def banner(draw, text, y, fg, bg):
  pad = 24
  f = _font(52, True)
  tw, th = draw.textbbox((0,0), text, font=f)[2:]
  bx = (W - (tw+pad*2))//2
  by = y
  draw.rectangle([bx, by, bx+tw+pad*2, by+th+pad], fill=bg)
  draw.text((bx+pad, by+pad//2), text, font=f, fill=fg)

def compose_frames(lines, brand="@aiprofithustlers", template="classic"):
  """Create 4 PNG frames (2s each) with bold OST. Returns list of file paths."""
  pal = neon_palette()
  frames = []

  # 3 panels + CTA panel
  panels = [lines["hook"], lines["value"], lines["proof"], lines["cta"]]
  colors = [pal["fg1"], pal["fg2"], pal["fg3"], pal["fg4"]]

  for i, panel_text in enumerate(panels):
    img = gradient_bg()
    d = ImageDraw.Draw(img)
    # Header banner
    header = {
      "classic": "REELS • 8s • AUTO-CONTENT",
      "urgent": "STOP SCROLLING",
      "educational": "QUICK TIP"
    }.get(template, "REELS • 8s • AUTO-CONTENT")
    banner(d, header, 72, fg=(0,0,0), bg=colors[i%len(colors)])

    # Main text
    wrapped = wrap_text(panel_text, width=16 if len(panel_text) > 40 else 12)
    draw_centered_text(d, wrapped, H//2 - 240, colors[i%len(colors)], size=98 if len(wrapped)<40 else 80, bold=True)

    # Brand footer
    if brand:
      fsm = _font(42, False)
      tw, th = d.textbbox((0,0), brand, font=fsm)[2:]
      d.text(((W-tw)//2, H - SAFE_MARGIN - th), brand, font=fsm, fill=pal["muted"])

    # Save PNG
    tmp = tempfile.NamedTemporaryFile(delete=False, suffix=f"_frame{i+1}.png")
    img.save(tmp.name, format="PNG")
    frames.append(tmp.name)

  return frames

def neon_palette():

def gradient_bg():
  p = neon_palette()
  img = Image.new("RGB", (W,H), p["bg_top"])

def banner(draw, text, y, fg, bg):

def centered(draw, text, y, color, size, bold=True):

  draw.multiline_text((x,y), text, font=f, align="center", fill=color)

def parse_prompt(p):
  p = p.strip()
  style = "promotional"
  if re.search(r"\beducation|teach|tutorial|how to|educational\b", p, re.I): style = "educational"
  if re.search(r"\bcontrarian|unpopular|hot take|myth\b", p, re.I): style = "contrarian"
  # niche/topic guess
  m = re.search(r"(for|about)\s+(.*)", p, re.I)
  niche = (m.group(2) if m else p)
  niche = niche[:120]
  return style, niche

def get_user_by_email(email: str) -> Optional[User]:

def ensure_user(email: str) -> User:

        u = s.exec(select(User).where(User.email==email)).first()
        if not u:
            u = User(email=email)
            s.add(u); s.commit(); s.refresh(u)
        return u

import os, jwt
from datetime import datetime, timedelta
from fastapi import Depends, HTTPException, Header
from pydantic import BaseModel

APP_SECRET = os.getenv("APP_SECRET","change-me")
JWT_TTL_MIN = int(os.getenv("JWT_TTL_MIN","10080"))

class TokenOut(BaseModel):
    token: str

def issue_token(email:str)->str:
    now = datetime.utcnow()
    payload = {"sub": email, "iat": int(now.timestamp()), "exp": int((now+timedelta(minutes=JWT_TTL_MIN)).timestamp())}
    return jwt.encode(payload, APP_SECRET, algorithm="HS256")

def require_user(authorization: str = Header(None)) -> str:
    if not authorization or not authorization.lower().startswith("bearer "):
        raise HTTPException(401, "Missing bearer token")
    token = authorization.split()[1]
    try:
        payload = jwt.decode(token, APP_SECRET, algorithms=["HS256"])
        return payload["sub"]
    except Exception:
        raise HTTPException(401, "Invalid token")

from fastapi import APIRouter, HTTPException
from pydantic import BaseModel
from sqlmodel import Session
from .db import engine, get_user_by_email

router = APIRouter(prefix="/gifts", tags=["gifts"])

class GiftSend(BaseModel):
    from_email: str
    to_email: str
    gift_value: int  # credits cost

@router.post("/send")
def send_gift(g: GiftSend):
    with Session(engine) as s:
        sender = get_user_by_email(g.from_email)
        receiver = get_user_by_email(g.to_email)
        if not sender or not receiver: raise HTTPException(404,"user")
        if sender.credits < g.gift_value: raise HTTPException(402,"insufficient credits")
        sender.credits -= g.gift_value
        # simplify: convert gifts to receiver credits 1:1 (later: wallet with rev-share)

import { useState, useEffect } from "react";
import Navbar from "./components/Navbar.jsx";
import Home from "./pages/Home.jsx";
import Pricing from "./pages/Pricing.jsx";
import Dashboard from "./pages/Dashboard.jsx";
import Creator from "./pages/Creator.jsx";
import Admin from "./pages/Admin.jsx";
import { login, me, subscribe } from "./api.js";

export default function App(){
  const [token,setToken]=useState(null);
  const [user,setUser]=useState(null);
  const [tab,setTab]=useState("home");
  const [email,setEmail]=useState("");

  useEffect(()=>{
    if(!token) return;
    me(token).then(setUser);
  },[token]);

  const onLogin = async ()=>{
    const r = await login(email);
    if(r.ok){ setToken(r.token); setUser(r.user); setTab("dashboard"); }
  };

  const onPlan = async (plan)=>{
    const r = await subscribe(user.email, plan);
    if(r.ok){ setUser(u=>({...u, plan:r.plan, credits:r.credits})); }
  };

  return (
    <>
      <Navbar user={user} onLogout={()=>{setToken(null); setUser(null); setTab("home");}} />
      <div style={{padding:"1rem"}}>
        {!user && (
          <div className="card" style={{marginBottom:"1rem"}}>
            <h3>Login</h3>
            <input placeholder="you@brand.com" value={email} onChange={e=>setEmail(e.target.value)} />
            <button className="btn" onClick={onLogin}>Enter</button>
          </div>
        )}

        <div style={{display:"flex",gap:8,marginBottom:12}}>
          <button className="btn" onClick={()=>setTab("home")}>Home</button>
          <button className="btn" onClick={()=>setTab("pricing")}>Pricing</button>
          {user && <button className="btn" onClick={()=>setTab("dashboard")}>Dashboard</button>}
          {user && <button className="btn" onClick={()=>setTab("creator")}>Creator</button>}
          <button className="btn" onClick={()=>setTab("admin")}>Admin</button>
        </div>

        {tab==="home" && <Home/>}
        {tab==="pricing" && <Pricing onChoose={onPlan}/>}
        {tab==="dashboard" && user && <Dashboard user={user}/>}
        {tab==="creator" && user && <Creator user={user}/>}
        {tab==="admin" && <Admin/>}
      </div>
    </>
  )
}
