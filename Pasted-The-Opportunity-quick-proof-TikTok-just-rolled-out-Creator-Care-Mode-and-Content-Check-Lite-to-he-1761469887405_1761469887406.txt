import os, re
from typing import List, Tuple
from fastapi import FastAPI, Request, Form
from fastapi.responses import HTMLResponse, RedirectResponse
from fastapi.staticfiles import StaticFiles
from fastapi.templating import Jinja2Templates

# Optional LLM (OpenAI-compatible). If no key, we fall back to heuristics.
from openai import OpenAI

OPENAI_API_KEY = os.getenv("OPENAI_API_KEY", "").strip()
OPENAI_BASE_URL = os.getenv("OPENAI_BASE_URL", "").strip()  # optional for self-hosted
OPENAI_MODEL = os.getenv("OPENAI_MODEL", "gpt-4o-mini")     # pick any available model

client = None
if OPENAI_API_KEY:
    client = OpenAI(api_key=OPENAI_API_KEY, base_url=OPENAI_BASE_URL or None)

app = FastAPI(title="TikTok Preflight")
app.mount("/static", StaticFiles(directory="static"), name="static")
templates = Jinja2Templates(directory="templates")

# --- Heuristic policy triggers (expandable) ---
BANNED_PATTERNS = [
    r"\b100%\b", r"\bguarantee(d)?\b", r"\bpermanent\b", r"\bultra\b", r"\bultimate\b",
    r"\b(get|earn)\s+\$\d+(?:,?\d+)*(?:\s+per\s+(day|week|month))?\b",
    r"\bmost\s+(effective|powerful)\b",
    r"\bfree\s+for\s+everyone\b",
    r"\b(before|after)\s+photos?\b",
    r"\bcure(s|d)?\b",
    r"\bweight\s*loss\b", r"\bmake\s+money\s+fast\b",
]

# Safer replacements (examples)
SOFTEN_MAP = {
    "100%": "highly likely",
    "guaranteed": "designed to help",
    "permanent": "long-lasting",
    "ultimate": "comprehensive",
    "cure": "may help improve",
    "free for everyone": "free for many users",
    "make money fast": "start earning sooner with effort",
}

SENSITIVE_SECOND_PERSON = [
    r"\b(you|your)\s*(?:(weight|income|health|appearance))",  # ‚Äúyour weight‚Äù, ‚Äúyour income‚Äù
]

def find_triggers(text: str) -> List[Tuple[str, str]]:
    hits = []
    for pat in BANNED_PATTERNS:
        for m in re.finditer(pat, text, flags=re.IGNORECASE):
            snippet = text[max(0, m.start()-20):m.end()+20]
            hits.append((pat, snippet))
    for pat in SENSITIVE_SECOND_PERSON:
        for m in re.finditer(pat, text, flags=re.IGNORECASE):
            snippet = text[max(0, m.start()-20):m.end()+20]
            hits.append((pat, snippet))
    return hits

def soften_text(text: str) -> str:
    out = text
    for k, v in SOFTEN_MAP.items():
        out = re.sub(rf"\b{k}\b", v, out, flags=re.IGNORECASE)
    # Avoid ‚Äúyou + sensitive attribute‚Äù
    out = re.sub(r"\byour (weight|income|health|appearance)\b",
                 r"people's \1", out, flags=re.IGNORECASE)
    # Tone-down absolutes
    out = re.sub(r"\b(always|never|best|#1)\b", "often", out, flags=re.IGNORECASE)
    return out

def llm_rewrite(prompt: str, system: str) -> str:
    if not client:
        return soften_text(prompt)
    msg = [
        {"role": "system", "content": system},
        {"role": "user", "content": prompt}
    ]
    resp = client.chat.completions.create(model=OPENAI_MODEL, messages=msg, temperature=0.4)
    return resp.choices[0].message.content.strip()

def gen_hooks(topic: str) -> List[str]:
    base_templates = [
        "Hook: What nobody tells {audience} about {topic}‚Ä¶",
        "Myth vs. Reality: {topic} in 15s",
        "I tested {topic} so you don‚Äôt have to‚Äîresults inside",
        "Before you post: 3 checks for {topic}",
        "Do this instead if {topic} keeps flopping",
    ]
    out = [t.format(audience="creators", topic=topic) for t in base_templates]
    if client:
        sys = ("You generate 5 short, compliant TikTok hooks (max 12 words, no absolutes,"
               " no personal attributes).")
        hooks = llm_rewrite(f"Topic: {topic}\nReturn list only.", sys)
        # Split lines if LLM used
        extra = [h.strip("-‚Ä¢ ").strip() for h in hooks.split("\n") if h.strip()]
        out = extra[:5] or out
    return out[:5]

def seo_keywords(topic: str) -> List[str]:
    # Simple seed + long-tails. LLM improves them if available.
    seeds = [
        f"{topic} tips", f"{topic} tutorial", f"{topic} for beginners",
        f"{topic} mistakes", f"{topic} checklist"
    ]
    if client:
        sys = ("Generate 8 TikTok search keywords for the topic. Keep 2‚Äì4 words each,"
               " no hashtags, no claims.")
        kws = llm_rewrite(f"Topic: {topic}", sys)
        seeds = [k.strip("-‚Ä¢ ").lower() for k in kws.split("\n") if k.strip()]
    return seeds[:8]

@app.get("/", response_class=HTMLResponse)
def home(request: Request):
    return templates.TemplateResponse("index.html", {"request": request, "result": None})

@app.post("/analyze", response_class=HTMLResponse)
def analyze(request: Request,
            topic: str = Form(...),
            text: str = Form(...)):
    triggers = find_triggers(text)
    safe_suggestion = llm_rewrite(
        f"Rewrite safely for TikTok ads/community guidelines. Keep meaning.\n\n{text}",
        ("You are a policy-aware copy editor. Remove absolutes, avoid sensitive attributes,"
         " keep claims soft and add disclaimers where apt.")
    )
    hooks = gen_hooks(topic)
    keywords = seo_keywords(topic)
    return templates.TemplateResponse("index.html", {
        "request": request,
        "result": {
            "triggers": triggers,
            "safe": safe_suggestion,
            "hooks": hooks,
            "keywords": keywords
        },
        "topic": topic,
        "text": text
    })
